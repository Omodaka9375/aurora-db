JavaScript
// Create a local database
const db = new AuroraDB({
    name: 'myapp',
    version: 1,
    keyPath: 'id'
});

// Initialize database
await db.create();

// Add data
await db.add('user1', {
    name: 'Alice Johnson',
    email: 'alice@example.com',
    role: 'Developer'
});

// Query data
const users = await db.filter(user => 
    user.role === 'Developer'
);

// Get all data
const allUsers = await db.all();

console.log('Found users:', users);
JavaScript
// Create distributed P2P database
const db = new AuroraDB({
    name: 'collaborative-app',
    distributed: true,
    autoSync: true,
    syncInterval: 10000,
    p2pAdapter: 'peerjs'
});

// Initialize with P2P sync
await db.create();

// Listen for sync events
db.addEventListener('sync', (event) => {
    console.log('Synced with peers:', event.detail);
});

// Listen for new peer connections
db.addEventListener('connection', (event) => {
    console.log('New peer connected:', event.detail.peerId);
});

// Add data - automatically syncs to peers
await db.add('task1', {
    title: 'Review code',
    assignee: 'Bob',
    priority: 'high',
    createdAt: new Date().toISOString()
});

// Data is now synchronized across all connected peers!
JavaScript
// Advanced configuration with custom conflict resolution
const db = new AuroraDB({
    name: 'enterprise-app',
    version: 2,
    distributed: true,
    syncStrategy: 'merge', // Custom conflict resolution
    indexes: [
        { name: 'email', unique: true },
        { name: 'department', unique: false },
        { name: 'createdAt', unique: false }
    ]
});

// Batch operations for performance
await db.transaction(async (db) => {
    db.entries['user1'] = { name: 'Alice', dept: 'Engineering' };
    db.entries['user2'] = { name: 'Bob', dept: 'Design' };
    db.entries['user3'] = { name: 'Charlie', dept: 'Marketing' };
});

// Advanced querying
const recentUsers = await db.filter(user => 
    new Date(user.createdAt) > new Date('2024-01-01')
);

// Pagination
const firstPage = await db.slice(0, 10);
const secondPage = await db.slice(10, 20);

// Complex analytics
const stats = {
    totalUsers: await db.count(),
    departments: await db.filter(u => u.department)
        .then(users => [...new Set(users.map(u => u.department))])
};

console.log('Database stats:', stats);

// Using different P2P adapters

// 1. PeerJS (Default)
const peerJSDB = new AuroraDB({
    name: 'peerjs-app',
    distributed: true,
    p2pAdapter: 'peerjs',
    server: {
        host: 'localhost',
        port: 9000,
        path: '/myapp'
    }
});

// 2. Custom WebRTC
const webrtcDB = new AuroraDB({
    name: 'webrtc-app',
    distributed: true,
    p2pAdapter: 'webrtc',
    p2pConfig: {
        signalingServer: 'wss://my-signaling.com',
        rtcConfig: {
            iceServers: [
                { urls: 'stun:stun.l.google.com:19302' }
            ]
        }
    }
});

// 3. Socket.IO
const socketDB = new AuroraDB({
    name: 'socket-app',
    distributed: true,
    p2pAdapter: 'socketio',
    p2pConfig: {
        serverUrl: 'https://my-socket-server.com',
        rooms: ['app-room']
    }
});

// 4. Custom Adapter
class MyCustomAdapter extends AuroraDB.P2PAdapter {
    async connect() {
        // Your custom connection logic
        console.log('Connecting with custom adapter');
    }
    
    send(peerId, data) {
        // Your custom send logic
        console.log('Sending data to', peerId);
    }
}

// Register and use custom adapter
AuroraDB.registerP2PAdapter('custom', MyCustomAdapter);

const customDB = new AuroraDB({
    name: 'custom-app',
    distributed: true,
    p2pAdapter: 'custom'
});
